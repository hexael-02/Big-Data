#declaramos la clase para calcular la edad de una persona.
#Unas clase es un mapa que definimos los atributos y metodos con el que seran creado los objetos 
#un atributo es una caracteristica (valores en variable) de un objeto
#Un metodo es el comportamineto de un objeto (las acciones o funciones que se pueden hacer)
# __init__: es la funcion constructora que se encarga de crear el objeto
# selt: es la variable global para acceder a la clase (se le eredan las caracteristicas del mismo)


class CalculadoraEdad:
	#la funcion "__init__" es la funcionn constructor que debe estar de primero 
	#la variable global "self" debe ir como parametro en cada metodo a definir en la clase

	def __init__(self, anio_nacimiento, mes_nacimiento, dia _nacimiento):
		self.fecha_nacimiento = date(anio_nacimiento, mes_nacimiento, dia_nacimiento)
	
	def calcular_edad (self):
		hoy = date.today()
		edad = hoy.year - self.fecha_nacimiento.year

		if (hoy.month,  hoy.day) < (self.fecha_nacimiento.month, self.fecha_nacimiento.day):
			edad -= 1

		return edad
